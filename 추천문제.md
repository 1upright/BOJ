## [2669. 직사각형 네개의 합집합의 면적 구하기](https://www.acmicpc.net/problem/2669)

```python
xa, ya, xb, yb = [], [], [], []

for _ in range(4):
    a, b, c, d = map(int, input().split())
    xa.append(a)
    ya.append(b)
    xb.append(c)
    yb.append(d)

N = max(ya + yb)
M = max(xa + xb)
arr = [[0]*M for _ in range(N)]

for k in range(4):
    for i in range(min(ya[k], yb[k]), max(ya[k], yb[k])):
        for j in range(min(xa[k], xb[k]), max(xa[k], xb[k])):
            arr[i][j] += 1

cnt = 0
for i in range(N):
    for j in range(M):
        if arr[i][j]:
            cnt += 1
print(cnt)
```

## [2116. 주사위 쌓기](https://www.acmicpc.net/problem/2116)

```python
def rev(num):
    if num == 0:
        return 5
    if num == 1:
        return 3
    if num == 2:
        return 4
    if num == 3:
        return 1
    if num == 4:
        return 2
    if num == 5:
        return 0

N = int(input())
arr = [list(map(int, input().split())) for _ in range(N)]
res = []
for j in range(6):
    a = j
    cnt = 0
    for i in range(N-1):
        cnt += max(list(set(arr[i]) - set([arr[i][a], arr[i][rev(a)]])))
        for k in range(6):
            if arr[i+1][k] == arr[i][rev(a)]:
                a = k
                break
    cnt += max(list(set(arr[N-1]) - set([arr[N-1][a], arr[N-1][rev(a)]])))
    res.append(cnt)
print(max(res))

## rev = {0: 5, 1: 3, 2: 4, 3: 1, 4: 2, 5: 0}으로 두었어야 했다
```

```python
cnt = 0
```

```python
cnt = [[0, 0] for _ in range(6)] # 1:동 2:서 3:남 4:북
for i in range(6):
    a, b = map(int, input().split())
    cnt[i] = [a, b]

max_col = 0

for i in range(6):
    if cnt[i][0] == 1 or cnt[i][0] == 2:
        if max_row < cnt[i][1]:
```

## [10157. 자리배정](https://www.acmicpc.net/problem/10157)

```python
# 시간 초과
di = [-1, 0, 1, 0] # 상 우 하 좌
dj = [0, 1, 0, -1]

def seat(n):
    if C*R < N:
        return [0]

    arr = [[0]*C for _ in range(R)]
    ni, nj = R-1, 0
    arr[ni][nj] = 1
    num = 1
    k = 0
    while 1:
        if 0<=ni+di[k%4]<R and 0<= nj+dj[k%4]<C and not arr[ni+di[k%4]][nj+dj[k%4]]:
            num += 1
            ni, nj = ni+di[k%4], nj+dj[k%4]
            arr[ni][nj] = num
            if num == N:
                return [nj+1, R-ni]
        else:
            k += 1

C, R = map(int, input().split())
N = int(input())
print(*seat(N))
```
